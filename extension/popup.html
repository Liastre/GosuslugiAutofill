Gosuslugi Autgofill v0.0.0

  • Расширение принимает на вход текстовый файл с данными в виде JSON, представление данных и все требуемые поля вы можете найти во вкладке "Формат".

    JSON данные можно выгрузить практически из любой базы данных, либо средствами систем учёта типа 1С.

  • Автозаполнение поиска

  • { "completed_work": [ // выполненные работы { "area": "область", // область "city": "город", // город "streetType": "ул.", // префикс улицы, ул/пер/пр-т "street": "название", // название улицы "house": "99", // номер дома "date": "01.2077", // дата в формате ММ.ГГГГ "services": [ // оказываемые услуги и выполняемые работы { "name": "", // наименование услуги "count": 0, // количество "cost": 0 // стоимость }, { ... // следующая услуга } ] } ] }
